38 {
39  math::Vector3 vector1_norm = vector1.Normalize();

44 double get_sin_angle(math::Vector3 reference_vector, math::Vector3 target_vector)
45 {
46  math::Vector3 reference_vector_norm = reference_vector.Normalize();
47  math::Vector3 target_vector_norm = target_vector.Normalize();
48  math::Vector3 cross_vector = reference_vector_norm.Cross(target_vector_norm);
49  return (cross_vector.z>0 ? cross_vector.GetLength() : -cross_vector.GetLength());
50 }

52 double get_angle_PI(math::Vector3 reference_vector, math::Vector3 target_vector) // return angle range [-PI,PI]
